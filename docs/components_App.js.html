/**
 * importing resources for react app
 * @import
 *
 * react - react app core resorces
 * react-markdown - for rendering markdown contents
 */
import React, { useState, useEffect } from "react";
import ReactMarkdown from "react-markdown";

/**
 * react component to render markdown contents
 * @component
 * @param {string} markdownContent - represent markdown contents
 * @returns {JSX.Element}  render markdown contents
 */
const MarkdownRenderer = ({ markdownContent }) => {
  return (
    <div className="markdown-container">
      <ReactMarkdown>{markdownContent}</ReactMarkdown>
    </div>
  );
};

/**
 * variable to store markdown contents
 * @type {string}
 */
const markdownContent = `
  #### My Dynaqmic Markdown Content

  This is a *dynamic* Markdown content example.

  - List item 1
  - List item 2

  [Learn more](https://example.com)
`;

/**
 * Functional component representing the main application.
 * @function
 * @component
 * @returns {JSX.Element} - render react main component
 */
const App = () => {
  /**
   * State to store the array of notes.
   * @type {Array<Object>}
   */
  const [notes, setNotes] = useState([]);

  /**
   * State to store the content of a new note.
   * @type {string}
   */
  const [newNote, setNewNote] = useState("");

  /**
   * Effect hook to fetch notes from the server on component mount.
   * @function
   * @name useEffect
   * @param {Function} effect - The function to execute.
   * @param {Array} dependencies - The dependencies to watch for changes.
   */
  useEffect(() => {
    fetch("/api/notes", { method: "GET" })
      .then((res) => res.json())
      .then((data) => setNotes(data.notes))
      .catch((err) => console.error(err));
  }, []);

  /**
   * Handler function to add a new note.
   * @function
   * @name handleAddNote
   */
  const handleAddNote = () => {
    fetch("/api/notes", {
      method: "POST",
      headers: {
        "Content-Type": "application/json",
      },
      body: JSON.stringify({ content: newNote }),
    })
      .then((res) => res.json())
      .then((data) => {
        console.log(data.message);
        setNewNote("");
        // Refresh notes after adding a new one
        fetch("/api/notes")
          .then((res) => res.json())
          .then((data) => setNotes(data.notes))
          .catch((err) => console.error(err));
      })
      .catch((err) => console.error(err));
  };

  /**
   * Render component.
   * @returns {JSX.Element} The JSX representation of the component.
   */
  return (
    <div>
      <h1>Express, React, and MongoDB</h1>
      <ul>
        {notes.map((note) => (
          <li key={note._id}>{note.content}</li>
        ))}
      </ul>
      <input
        type="text"
        placeholder="Add a new note"
        value={newNote}
        onChange={(e) => setNewNote(e.target.value)}
      />
      <button onClick={handleAddNote}>Add Note</button>

      <div>
        <h3>React Markdown Renderer</h3>
        <MarkdownRenderer markdownContent={markdownContent} />
      </div>
    </div>
  );
};

export default App;
